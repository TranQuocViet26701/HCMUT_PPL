
        Class Sothu{
            Var Thuc, Ao: Int;
            Constructor (Thuc, Ao: Int){
                Self.Thuc = Thuc;
                Self.Ao = Ao;
            }
            Add(a: Sothuc){
                Self.Thuc = a.Thuc + Self.Thuc;
                Self.Ao = a.Ao + Self.Ao;
            }
            }
