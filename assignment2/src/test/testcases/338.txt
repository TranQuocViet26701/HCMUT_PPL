
        Class Shape {
            Val length,width: Int;
            getArea(){ }
            Shape(length,width: Float){
                Self.length = length;
                Self.width = width;
            }
        }

        Class Rectangle: Shape {
            getArea(){
                Return Self.length * Self.width;
            }
        }
        Class Triangle: Shape {
            getArea(){
                Return Self.length * Self.width / 2;
            }
        }
