
        Class a{
            bubbleSort(arr: Array[Int,1000]; len: Int){
                Var i,j: Int;
                Foreach (i In 0 .. len){
                    Foreach (j In len-1 .. i){
                        If(arr[j] < arr[j-1]){
                            temp = arr[j];
                            arr[j] = arr[j-1];
                            arr[j-1] = temp;
                        }
                    }
                }
                Return arr;
            }
        }
