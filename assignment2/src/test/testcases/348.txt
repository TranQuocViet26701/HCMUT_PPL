
    Class PlayerMoment : MonoBehaviour{

        Var speed : Int = 12;
        Val  gravity : Float = -9.81* 10;
        ## Update is called once per frame ##
         Update()
        {
            isGrounded = Physics.CheckSphere(groundCheck::$position, groundDistance, groundMask);

            If (isGrounded && velocity.y < 0) {
                velocity.y = -2;
            }
        }
    }
