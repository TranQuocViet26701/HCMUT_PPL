
        Class BinExp : Exp {
            Val op : String;
            Val left, right : Float;
            BinExp(op : String; l, r : Float) {
                Self.op = op;
                Self.r = r;
            }
            eval() {
                If (op == "+") {Return left + right;}
                Elseif (op == "-") {Return lef - right;}
                Elseif (op == "/") {Return lef / right;}
            }
        }
